\SACCMD{enddevices}
\label{cmd:enddevices}

\SACTitle{概要}
结束某个图像设备

\SACTitle{语法}
\begin{SACSTX}
B!EGIN!D!EVICES! S!GF!|X!WINDOWS!
\end{SACSTX}

\SACTitle{输入}
\begin{itemize}
\item SGF : SAC图形文件设备驱动
\item XWINDOWS : X Window System图像窗口系统
\end{itemize}

\SACTitle{说明}
参见命令begindevices的说明。

\SACTitle{相关命令}
\nameref{cmd:begindevices}
