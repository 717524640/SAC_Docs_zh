\SACCMD{log10}
\label{cmd:log10}

\SACTitle{概要}
对每个数据点做以10为底对数

\SACTitle{语法}
LOG10

\SACTitle{错误消息}
\begin{itemize}
\item[-]1301: 未读入文件
\item[-]1307: 对谱文件非法操作
\item[-]1340: 文件中数据点值超出允许的范围（对数要求值为正）
\end{itemize}

\SACTitle{头段变量改变}
DEPMIN, DEPMAX, DEPMEN

\SACTitle{最近修订}
January 15, 1985 (Version 9.10)
